* Setting up podman containers

#+begin_src shell
mkdir ~/.local/containers/mumble/data-volume

podman volume create \
           --opt type=none \
           --opt device=$HOME/.local/containers/mumble/data-volume/ \
           --opt o=bind \
           mumble-data

podman pod create --name botcodes \
           -p 8040:80 \
           -p 8041:443 \
           -p 8042:2222 \
           -p 8007:8007/tcp \
           -p 8007:8007/udp \

           --name nginx \
