*** Mumble

  #+begin_src shell
    mkdir ~/.local/containers/mumble/data-volume

    podman volume create \
           --opt type=none \
           --opt device=$HOME/.local/containers/mumble/data-volume/ \
           --opt o=bind \
           mumble-data

    podman run -d \
           --name mumble-server \
           --pod=botcodes \
           -v mumble-data:/data:Z \
           -v $HOME/.local/containers/letsencrypt:/etc/letsencrypt:ro \
           -e "MUMBLE_CONFIG_PORT=8007" \
           docker.io/mumblevoip/mumble-server

    #+end_src
